# test parameters could be specified for the test run
# they are passed during the initialization phase to the server
# default = no parameters
# Currently supported parameters:
#   ResetExcluded:mmm[,nnn...]  specifies list of internal subtest IDs,
#                               whose data must not be reset
#   ResetIncluded:mmm[,nnn...]  specifies list of internal subtest IDs,
#                               whose data must be reset
#   In both cases the IDs must belong to subtests of the test class
#   specified below.
#   These parameters could be used in combination with the property 'resettestdata = 1'
#   as well as with the special action 'ResetTestData'. If resettestdata=0 and
#   special action 'ResetTestData' is not set, ResetExcluded resp. ResetIncluded will be ignored.
#   It is not allowed to set both parameters together. 
# syntax: testparameters=ResetExcluded:mmm[,nnn...] resp. ResetIncluded:mmm[,nnn...]
# testparameters = ...

###################################
#     Special actions             #
###################################

# no test will be executed, but the client driver should execute a special task
# currently possible actions:
# - CleanAdminData  is used if on the server side it is erroneously indicated that a test is running on the test class;
#                   this scenario could happen, if the server was restarted during a test run and was not up and running,
#                   when the test finished on client side; therefore the administration data of the test run could not be adapted
#                   in the database on the server side;
#                   other potential causes are severe crashes on the client or server side during a test run;
#                   'CleanAdminData' is used to put the test administration data back to a consistent state;
#                   no test run is executed; therefore the properties for test and measurement information are not needed;
#                   use this action only, if you are sure that there is really no test running on server side, but the server reports
#                   that a test is running for this class;
#                   because this action is a severe intervention, it has to be confirmed during the run
# - ResetTestData   is used if data of old test runs should be deleted without executing a new test;
#                   reset of data is restricted to runs of the test class specified below and could be further limited
#                   by setting the property testparameters=ResetExcluded or ...=ResetIncluded (see above) 
#specialaction = CleanAdminData

# Following kinds of authentication are supported:
# - server authentication
#   trust store is used (if not set, JRE default is used)
#   - with user/password
#   - w/o user/password
#     user/password has to be set on server side in transaction SICF for service 'hatool'
# - mutual authentication
#   trust store and key store are used

## trust store ##
# used for root certifcate of the server to allow server authentication
# default: <JRE_HOME>/lib/security/cacerts
# could be provided as java property on the command line:
#   -Djavax.net.ssl.trustStore=<file>
# or as HA Test Tool property;
# Setting both, the Java as well as the HA Test Tool property, results in a 'Parameter declared multiple times' error.
# httpstruststore = <file name>

# As password for the trust store, the default of cacerts can be used (no extra specification necessary)
# or it could be provided as java property on the command line:
#  -Djavax.net.ssl.trustStorePassword=<psw>
# or as HA Test Tool property;
# Setting both, the Java as well as the HA Test Tool property, results in a 'Parameter declared multiple times' error.
# httpstruststorepassword = xxxxxxxxx

## key store ##
# contains the client certificate in PKCS12 format
# is only needed if mutual authentication should be done;
# could be provided as java property on the command line:
#   -Djavax.net.ssl.keyStore=<file>
# or as HA Test Tool property;
# Setting both, the Java as well as the HA Test Tool property, results in a 'Parameter declared multiple times' error.
# httpskeystore = <file name>

# The password for the key store could be provided as java property on the command line:
#  -Djavax.net.ssl.keyStorePassword=<psw>
# or as HA Test Tool property;
# Setting both, the Java as well as the HA Test Tool property, results in a 'Parameter declared multiple times' error.
